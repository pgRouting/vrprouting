74 static
75 void db_get_breaks(
76  char *breaks_sql,
77  Vroom_break_t **breaks,
78  size_t *total_breaks,
79 
80  Column_info_t *info,
81  const int column_count,
82  bool is_plain) {
83 #ifdef PROFILE
84  clock_t start_t = clock();
85  PGR_DBG("%s", breaks_sql);
86 #endif
87 
88  const int tuple_limit = 1000000;
89 
90  size_t total_tuples;
91 
92  void *SPIplan;
93  SPIplan = pgr_SPI_prepare(breaks_sql);
94  Portal SPIportal;
95  SPIportal = pgr_SPI_cursor_open(SPIplan);
96 
97  bool moredata = true;
98  (*total_breaks) = total_tuples = 0;
99 
100  /* on the first tuple get the column numbers */
101 
102  while (moredata == true) {
103  SPI_cursor_fetch(SPIportal, true, tuple_limit);
104  if (total_tuples == 0) {
105  pgr_fetch_column_info(info, column_count);
106  }
107  size_t ntuples = SPI_processed;
108  total_tuples += ntuples;
109  if (ntuples > 0) {
110  if ((*breaks) == NULL)
111  (*breaks) = (Vroom_break_t *)palloc0(
112  total_tuples * sizeof(Vroom_break_t));
113  else
114  (*breaks) = (Vroom_break_t *)repalloc(
115  (*breaks),
116  total_tuples * sizeof(Vroom_break_t));
117 
118  if ((*breaks) == NULL) {
119  elog(ERROR, "Out of memory");
120  }
121 
122  size_t t;
123  SPITupleTable *tuptable = SPI_tuptable;
124  TupleDesc tupdesc = SPI_tuptable->tupdesc;
125  for (t = 0; t < ntuples; t++) {
126  HeapTuple tuple = tuptable->vals[t];
127  fetch_breaks(&tuple, &tupdesc, info,
128  &(*breaks)[total_tuples - ntuples + t], is_plain);
129  }
130  SPI_freetuptable(tuptable);
131  } else {
132  moredata = false;
133  }
134  }
135 
136  SPI_cursor_close(SPIportal);
137 
138  if (total_tuples == 0) {
139  (*total_breaks) = 0;
140  return;
141  }
142 
143  (*total_breaks) = total_tuples;
144 #ifdef PROFILE
145  time_msg("reading breaks", start_t, clock());
146 #endif
147 }

180  }
