85 void
86 Pgr_pickDeliver::solve() {
87  auto initial_sols = solutions;
88 
89  if (m_initial_id == 0) {
90  msg.log << "trying all \n";
91  for (int i = 1; i < 7; ++i) {
92  initial_sols.push_back(Initial_solution((Initials_code)i, m_orders.size()));
93  msg.log << "solution " << i << "\n" << initial_sols.back().tau();
94  // TODO(vicky) calculate the time it takes
95  msg.log << "Initial solution " << i
96  << " duration: " << initial_sols.back().duration();
97  }
98  } else {
99  msg.log << "only trying " << m_initial_id << "\n";
100  initial_sols.push_back(Initial_solution((Initials_code)m_initial_id, m_orders.size()));
101  // TODO(vicky) calculate the time it takes
102  msg.log << "Initial solution " << m_initial_id
103  << " duration: " << initial_sols[0].duration();
104  }
105 
106 
107  /*
108  * Sorting solutions: the best is at the back
109  */
110  pgassert(!initial_sols.empty());
111  std::sort(initial_sols.begin(), initial_sols.end(), []
112  (const Solution &lhs, const Solution &rhs) -> bool {
113  return rhs < lhs;
114  });
115 
116 #if 1
117  solutions.push_back(Optimize(initial_sols.back(), m_max_cycles));
118 #else
119  solutions.push_back(initial_sols.back());
120 #endif
121  pgassert(!solutions.empty());
122 
123  msg.log << "best solution duration = " << solutions.back().duration();
124 }

128 std::vector< General_vehicle_orders_t >
129 Pgr_pickDeliver::get_postgres_result() const {
130  auto result = solutions.back().get_postgres_result();
131 
132  General_vehicle_orders_t aggregates = {
133  /*
134  * Vehicle id = -2 indicates its an aggregate row
135  *
136  * (twv, cv, fleet, wait, duration)
137  */
138  -2, // summary row on vehicle_number
139  solutions.back().twvTot(), // on vehicle_id
140  solutions.back().cvTot(), // on vehicle_seq
141  -1, // on order_id
142  -1, // on stop_id
143  -2, // on stop_type (gets increased later by one so it gets -1)
144  -1, // not accounting total loads
145  solutions.back().total_travel_time(),
146  -1, // not accounting arrival_travel_time
147  solutions.back().wait_time(),
148  solutions.back().total_service_time(),
149  solutions.back().duration(),
150  };
151  result.push_back(aggregates);
152 
153 
154 #ifndef NDEBUG
155  for (const auto &sol : solutions) {
156  msg.log << sol.tau();
157  }
158 #endif
159  return result;
160 }
