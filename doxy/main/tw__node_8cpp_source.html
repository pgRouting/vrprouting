93 bool
94 Tw_node::is_compatible_IJ(const Tw_node &I, Speed speed) const {
95  /*
96  * I /-> kStart
97  */
98  if (m_type == kStart) return false;
99 
100  /*
101  * kEnd /-> (*this)
102  */
103  if (I.m_type == kEnd) return false;
104 
105  return !is_late_arrival(arrival_j_opens_i(I, speed));
106 }
